1. Membuat Database

CREATE DATABASE library;

2. Membuat Table di Dalam Database

CREATE TABLE users (
    id int AUTO_INCREMENT,
    name varchar(255),
    email varchar(255),
    password varchar(255),
    PRIMARY KEY (id)
);

CREATE TABLE categories (
    id int AUTO_INCREMENT,
    name varchar(255),
    PRIMARY KEY (id)
);

CREATE TABLE books (
    id int AUTO_INCREMENT,
    title varchar(255),
    summary text,
    stock int,
    categoryID int,
    PRIMARY KEY (id),
    FOREIGN KEY (categoryID) REFERENCES categories(id)
);
